\newenvironment{rounded_frame}[5]
{
    \mdfsetup{
      middlelinecolor=mybluegrey,
      middlelinewidth=1pt,
      backgroundcolor=white!0,
      roundcorner=10pt,
      skipabove=30,
      fontcolor=black,
      innertopmargin=-10pt,
      innerrightmargin=17pt,
    }
    \begin{internal}{#1}{#2}{#3}{#4}{#5}
}
{
    \end{internal}
}
